// --- MedicalRecordService.cs ---
using proekt.Data;
using proekt.Models;
using Microsoft.EntityFrameworkCore;
using System.Text.Json;

namespace proekt.Services;

public class MedicalRecordService {
    private readonly ApplicationDbContext _db;
    public MedicalRecordService(ApplicationDbContext db) { _db = db; }

    public MedicalRecord CreateEmptyRecord(int userId) {
        var existing = _db.MedicalRecords.FirstOrDefault(r => r.UserId == userId);
        if (existing != null) return existing;
        var record = new MedicalRecord { UserId = userId, CreatedAt = DateTime.Now, Entries = new List<MedicalEntry>() };
        _db.MedicalRecords.Add(record);
        _db.SaveChanges();
        return record;
    }

    public MedicalRecord? GetRecordByUserId(int userId)
        => _db.MedicalRecords.Include(r => r.Entries.Where(e => !e.IsDeleted)).FirstOrDefault(r => r.UserId == userId);

    public MedicalEntry AddEntry(int userId, MedicalEntryType type, string title, string? description, DateTime date, int doctorId, string doctorName) {
        var record = GetRecordByUserId(userId) ?? CreateEmptyRecord(userId);
        var entry = new MedicalEntry { MedicalRecordId = record.Id, Type = type, Title = title, Description = description, Date = date, CreatedByDoctorId = doctorId, CreatedByDoctorName = doctorName, CreatedAt = DateTime.Now };
        _db.MedicalEntries.Add(entry);
        _db.SaveChanges();
        AddAuditLog(record.Id, entry.Id, "AddEntry", doctorId, doctorName, $"Added {type}: {title}");
        return entry;
    }

    public void UpdateRecordInfo(int userId, string? bloodType, string? allergies, string? chronicConditions, string? generalNotes, int actorId, string actorName) {
        var record = _db.MedicalRecords.FirstOrDefault(r => r.UserId == userId);
        if (record == null) return;
        var oldSnap = JsonSerializer.Serialize(new { record.BloodType, record.Allergies, record.ChronicConditions, record.GeneralNotes });
        record.BloodType = bloodType; record.Allergies = allergies; record.ChronicConditions = chronicConditions; record.GeneralNotes = generalNotes;
        _db.SaveChanges();
        var newSnap = JsonSerializer.Serialize(new { record.BloodType, record.Allergies, record.ChronicConditions, record.GeneralNotes });
        AddAuditLogWithSnapshot(record.Id, null, "UpdateRecord", actorId, actorName, "Updated general record info", oldSnap, newSnap);
    }

    private void AddAuditLog(int recordId, int? entryId, string actionType, int doctorId, string doctorName, string details) {
        _db.MedicalAuditLogs.Add(new MedicalAuditLog { MedicalRecordId = recordId, EntryId = entryId, ActionType = actionType, DoctorId = doctorId, DoctorName = doctorName, Timestamp = DateTime.Now, Details = details });
        _db.SaveChanges();
    }

    private void AddAuditLogWithSnapshot(int recordId, int? entryId, string actionType, int actorId, string actorName, string details, string? oldSnap, string? newSnap) {
        _db.MedicalAuditLogs.Add(new MedicalAuditLog { MedicalRecordId = recordId, EntryId = entryId, ActionType = actionType, DoctorId = actorId, DoctorName = actorName, Timestamp = DateTime.Now, Details = details, OldSnapshot = oldSnap, NewSnapshot = newSnap });
        _db.SaveChanges();
    }

    public List<MedicalAuditLog> GetAuditLogs(int recordId)
        => _db.MedicalAuditLogs.Where(l => l.MedicalRecordId == recordId).OrderByDescending(l => l.Timestamp).ToList();
}

// --- MedicalRecordController.cs ---
using Microsoft.AspNetCore.Mvc;
using proekt.Models;
using proekt.Models.ViewModels;
using proekt.Services;

namespace proekt.Controllers;

public class MedicalRecordController : Controller {
    private readonly MedicalRecordService _medRecordService;
    private readonly UserService _userService;

    public MedicalRecordController(MedicalRecordService medRecordService, UserService userService) {
        _medRecordService = medRecordService;
        _userService = userService;
    }

    [HttpGet]
    public IActionResult ViewRecord(int userId) {
        if (HttpContext.Session.GetString("UserEmail") == null) return RedirectToAction("Login", "Home");
        var record = _medRecordService.GetRecordByUserId(userId) ?? _medRecordService.CreateEmptyRecord(userId);
        var patient = _userService.GetUserById(userId);
        if (patient == null) return NotFound();
        var vm = new MedicalRecordViewModel { Patient = patient, Record = record, AuditLogs = _medRecordService.GetAuditLogs(record.Id), CanEdit = (HttpContext.Session.GetString("UserRole") == "Doctor"), CanEditBasicInfo = true };
        return View(vm);
    }

    [HttpPost]
    [ValidateAntiForgeryToken]
    public IActionResult AddEntry(int userId, string type, string title, string? description, string date) {
        if (!Enum.TryParse<MedicalEntryType>(type, out var entryType)) entryType = MedicalEntryType.Note;
        _medRecordService.AddEntry(userId, entryType, title, description, DateTime.Parse(date), 1, "System Doctor");
        return RedirectToAction("ViewRecord", new { userId });
    }
}

using Microsoft.AspNetCore.Http;
namespace proekt.Services;

public class TranslationService {
    private readonly IHttpContextAccessor _http;
    private const string SessionKey = "Lang";
    private readonly Dictionary<string, Dictionary<string, string>> _data;

    public TranslationService(IHttpContextAccessor http) {
        _http = http;
        _data = new Dictionary<string, Dictionary<string, string>>(StringComparer.OrdinalIgnoreCase) {
            ["en"] = new Dictionary<string, string> { ["Home"] = "Home", ["Login"] = "Login", ["Register"] = "Register", ["EHRSystem"] = "EHR System" /* ... 200+ keys ... */ },
            ["bg"] = new Dictionary<string, string> { ["Home"] = "Начало", ["Login"] = "Вход", ["Register"] = "Регистрация", ["EHRSystem"] = "EHR Система" /* ... 200+ keys ... */ }
        };
    }
    public string T(string key) {
        var lang = _http.HttpContext?.Session.GetString(SessionKey) ?? "en";
        return _data.ContainsKey(lang) && _data[lang].ContainsKey(key) ? _data[lang][key] : key;
    }
}

    [Allergies] NVARCHAR(MAX),
    [ChronicConditions] NVARCHAR(MAX),

    [NewSnapshot] NVARCHAR(MAX),

1. УВОД

1.1. Предисловие и актуалност на избраната тема

В зората на третото десетилетие на XXI век човечеството е изправено пред предизвикателства, които изискват не просто адаптация, а пълно преосмисляне на начина, по който функционират критичните публични сектори. Здравеопазването, като най-важен стълб на човешкото благополучие, се намира в центъра на тази трансформация. Избраната тема за дипломен проект — „Медицински портал за електронни досиета, консултации и управление на здравни данни (MedReports)“ — е не само актуална, но и критично необходима в условията на глобална дигитализация.

Актуалността на проекта се корени в няколко фундаментални проблема на традиционната медицина, които се наблюдават както в глобален мащаб, така и със специфична тежест в българската реалност. Първият и най-видим проблем е фрагментацията на данните. В традиционния модел информацията за здравето на един човек е разпръсната в десетки различни медии — хартиени амбулаторни листове, съхранявани в личните архиви на лекарите, частни лаборатории, които издават резултати само на хартия или в затворени онлайн портали, и болнични архиви, които често са недостъпни извън самото лечебно заведение. Тази разпокъсаност прави невъзможно съставянето на пълна клинична картина за секунди, което е решаващо в спешни ситуации, където всяка секунда е равна на живот.

Вторият аспект на актуалността е бюрократичната тежест. Лекарите са натоварени с огромно количество административна работа, която буквално "краде" от времето за диагностика и внимание към пациента. Дигиталното решение MedReports цели да съкрати времето за попълване на диагнози и рецепти от минути до секунди чрез автоматизирани подсказки и шаблони. 

Третият ключов елемент е липсата на контрол от страна на пациента. В сегашната система пациентът рядко има пълен и моментален достъп до собствените си изследвания и история. Той често е в ролята на "пощальон", който пренася папки с документи от кабинет на кабинет. Проектът MedReports обръща тази парадигма, поставяйки пациента в центъра на системата, като му предоставя пълен достъп и контрол над собствените му данни.

Когато един лекар разполага с пълната хронология на алергиите, хроничните заболявания и предходните лечения на пациента, вероятността за медицинска грешка — която по данни на СЗО е една от водещите причини за смъртност в света — намалява драстично. В този смисъл, MedReports не е просто база данни, а инструмент за спасяване на човешки живот.

1.2. Цели и задачи на дипломния проект

Основната цел на настоящата разработка е проектирането, реализирането и тестването на мащабируема, сигурна и интуитивна софтуерна система за управление на здравни досиета, която да служи за модел на модерна e-Health инфраструктура. Системата трябва да осигурява надеждна връзка между лекаря и пациента, като гарантира интегритета на информацията.

За постигането на тази голяма цел са дефинирани следните комплексни задачи, разпределени в четири основни направления:

Проучвателна задача:
- Анализ на международните стандарти за съхранение на медицински данни (HL7 FHIR, EDI) и тяхното адаптиране към българските условия.
- Изследване на европейската правна рамка за защита на личните данни (GDPR) и специфичните изисквания към медицинската тайна съгласно българския Закон за здравето.
- Сравнителен анализ на съществуващи софтуерни решения на пазара, идентифициране на техните силни страни и пропуски в потребителското изживяване.
- Проучване на софтуерните архитектури, подходящи за високо натоварени системи с чувствителна информация.

Проектантска задача:
- Създаване на релационен модел на база данни, оптимизиран за бързо търсене по индекси и минимално дублиране на данни (Normal Form 3).
- Проектиране на софтуерна архитектура по модела „Service-Oriented Architecture (SOA)“, която да позволи лесно разделяне на отговорностите и бъдещо преминаване към микросървисна структура.
- Разработване на UI/UX палитра и навигационна структура, която да е достъпна за хора от всички възрастови групи, включително за такива с ниска степен на дигитална грамотност.

Имплементационна задача:
- Изграждане на ядрото на приложението с ASP.NET Core 8.0, използвайки C# като съвременен, типизиран и високопроизводителен език.
- Реализиране на ролево-базиран достъп (RBAC: Admin, Doctor, Patient), гарантиращ, че данните са "locked down" по подразбиране.
- Разработване на механизъм за „Medical Snapshot Auditing“, който записва пълната история на промените в здравословното състояние на пациента под формата на версии.
- Интегриране на сигурно файлово хранилище за прикачване на бинарни документи (PDF рентгенови снимки, лабораторни анализи).

Верификационна задача:
- Извършване на функционално тестване на всички модули - от регистрацията до генерирането на медицински записи.
- Провеждане на нефункционални тестове за скорост на базата данни и устойчивост на атаки (Cross-site Scripting, SQL Injection).
- Симулация на потребителски сценарии за оценка на UX дизайна и леснотата на навигация.

1.3. Социално-икономическо значение на e-Health сектора

Внедряването на цифрови решения в здравеопазването не е просто технологичен въпрос, а икономическа необходимост. Дигитализацията води до:
- Оптимизация на ресурсите: Намаляване на дублираните изследвания. Често пациентите правят едни и същи скъпи кръвни тестове в различни болници, само защото лекарите нямат общ достъп до информацията.
- Превенция: Редовното следене на здравните показатели през портала позволява ранно диагностициране на хронични заболявания, чието лечение в ранен стадий е в десетки пъти по-евтино за държавния бюджет.
- Достъпност: Телемедицината и лесният достъп до досието позволяват на хора в отдалечени региони да споделят своите данни с топ специалисти, без нужда от скъпи пътувания.

От социална гледна точка, MedReports повишава общата дигитална култура на населението в контекста на здравето. Когато един човек вижда своята кръвна група, алергии и имунизационен календар на екрана на телефона си, той става по-отговорен към собственото си здравословно състояние. Това е стъпка към "активното здравеопазване", където пациентът не е пасивен получател на грижи, а активен партньор на лекаря.

1.4. Глобалните цели на ЕС за "Цифрово десетилетие 2030"

Европейският съюз в своята стратегия "Цифров компас" постави ясна цел: До 2030 година 100% от европейските граждани трябва да имат онлайн достъп до своите електронни здравни досиета. MedReports отговаря на този призив, като изгражда инфраструктура, подготвена за "Cross-border" обмен на данни. Използвайки JSON като основен формат за представяне на данните в нашия API слой, системата е готова да комуникира в бъдеще с други европейски медицински мрежи, позволявайки на български гражданин да бъде лекуван адекватно навсякъде в ЕС, благодарение на неговото MedReports досие.

Европа също така набляга на "Суверенитета на данните". Това означава, че данните на българите трябва да се съхраняват в сигурни европейски облачни структури, а не в неконтролирани външни сървъри. Проектът е проектиран с "Cloud-Agnostic" дизайн, което ни позволява да го разгърнем на всеки сертифициран облачен провайдър, отговарящ на изискванията на ЕС.

1.5. EHR срещу PHR – видове електронни здравни записи

В медицинската информатика съществува важно разделение, което е залегнало в основите на нашия проект:
- EHR (Electronic Health Record): Това са официалните записи на лечебното заведение (болница, ДКЦ). Те са ориентирани към медицинската практика и се попълват само от лекари.
- PHR (Personal Health Record): Това е записът, който се контролира лично от пациента. Пациентът може да добавя информация за ежедневни симптоми, диета и физическа активност.

MedReports е хибридна система, която съчетава сигурността на EHR с гъвкавостта на PHR. Лекарят попълва официалните диагнози, които имат правна тежест, но пациентът също има право да добавя "Лични бележки" и свои документи. Този "хибриден модел" се счита за най-ефективния начин за изграждане на доверие в медицинския софтуер, тъй като пациентът вижда, че не е просто "номер в таблица", а неговото лично мнение и наблюдения също са част от системата.

1.6. Структура на дипломната работа

Настоящият дипломен проект е структуриран в няколко логически последователни части:
- Глава 1 (Проучвателна): Разглежда историята на технологиите в медицината, нормативната уредба в България и ЕС, и прави избор на технологичен стек.
- Глава 2 (Изисквания): Дефинира детайлно какво трябва да прави системата и на какви стандарти за качество трябва да отговаря.
- Глава 3 (Техническа имплементация): Най-обемната част, описваща архитектурата, базата данни, бекенд логиката с C# и фронтенд решенията с Razor.
- Глава 4 (Ръководство): Практически насоки за инсталация, ползване и поддръжка, включително и QA анализ.
- Приложения: Съдържат сорс кода, SQL схемите и библиографията.

Тази структура е избрана така, че да води читателя от теоретичните основи до конкретния програмен код, демонстрирайки пълния цикъл на софтуерна разработка (SDLC).

2. ИЗЛОЖЕНИЕ

2.1. Глава 1 Проучвателна част

2.1.1. Исторически преглед и еволюция на e-Health системите

Дигитализацията в медицината не започва днес. Нейните корени могат да бъдат проследени до 60-те години на миналия век, когато за първи път в университетските болници в САЩ са използвани големи мейнфрейм компютри за автоматизиране на счетоводните процеси. По това време софтуерът е бил чисто административен — той е следял инвентара на лекарствата и плащанията на пациентите.

През 70-те години започват първите опити за създаване на клинични системи за подпомагане на вземането на решения (Clinical Decision Support Systems). Една от най-известните системи от този период е MYCIN, разработена в Станфорд, която е използвала ранен изкуствен интелект (база от правила), за да помага на лекарите при диагностика на бактериални инфекции. Въпреки технологичните успехи, липсата на мрежова свързаност е ограничавала тези системи само до един компютър.

2.1.2. Революцията на персоналните компютри и мрежите

Истината за e-Health се промени радикално през 90-те години с появата на интернет. Уеб браузърът стана универсалният прозорец към медицинската информация. Вече не беше необходимо лекарят да е в същата стая с компютъра, съдържащ данните. Възможността за комуникация през протоколи като HTTP трансформира болницата от затворена "крепост на знания" в отворена "мрежа от данни".

Проектът MedReports стъпва върху тези исторически уроци. Ние не създаваме софтуер за един компютър, а за мрежата. Нашата архитектура използва RESTful принципите за комуникация, което е естественото продължение на еволюцията от затворени мрежи към облачни портали.

2.1.3. Цифрова трансформация в ЕС и България (НЗИС)

България е в процес на активна цифрова трансформация в здравеопазването. Изграждането на НЗИС (Национална здравноинформационна система) е огромен държавен проект, който въведе електронната рецепта и електронното направление. Въпреки това, държавните системи често страдат от тромав интерфейс и липса на персонализация за частния сектор.

MedReports се явява като "дигитален слой", който може да работи паралелно или да се интегрира в бъдеще, като предлага много по-високо ниво на UX (User Experience). Докато държавната система е ориентирана към отчетността пред НЗОК, MedReports е ориентирана към пациента и неговия комфорт. Ние вярваме, че бъдещето е в колаборацията между държавни стандарти и частни иновативни платформи.

2.1.4. Българската нормативна уредба (Закон за здравето/ЗЗЛД)

Работата с медицински данни е строго регулирана в България:
- Закон за здравето: Дефинира правата и задълженията на лекаря и пациента. Член 27 забранява разпространението на лични здравни данни без съгласие, освен в изключителни случаи.
- ЗЗЛД (Закон за защита на личните данни): Този закон регламентира как софтуерните фирми трябва да съхраняват информацията.
MedReports внедрява т.нар. "Privacy by Default". Пациентът сам решава дали да даде достъп до своето досие на конкретен лекар. Тази "гранулярност" на достъпа е гаранция за спазване на законите и професионалната етика.

2.1.5. Регламент (ЕС) 2016/679 (GDPR) в медицинския сектор

GDPR класифицира медицинските данни като "специална категория данни" (Sensitive Data), които изискват засилена защита. MedReports прилага трите стълба на GDPR:
1. Право на достъп: Пациентът вижда всичко, което е записано за него.
2. Право на изтриване (Забрава): Въпреки че законодателството ограничава изтриването на официални медицински записи, MedReports позволява анонимизиране на личните профили.
3. Сигурност на обработката: Шифроване на данните при съхранение и транзит. Всички заявки в портала минават през HTTPS/TLS 1.3.

2.1.6. Сравнителен анализ на съществуващи софтуерни решения

В процеса на разработка бяха анализирани няколко водещи системи на българския пазар:
- Superdoc: Лидер в записването на часове, но на практика не поддържа медицински досиета. Тя е транзакционна система за графици.
- Healee: Иновативна платформа за телемедицина, но интерфейсът е изключително сложен за по-възрастни потребители.
- Consento: Силна в B2B сектора (интеграция с фирми), но затворена за индивидуалния свободен избор на лекари.

MedReports запълва празнината чрез "User-Centric" дизайн. Ние се фокусираме върху три клика: Вход -> Избор на лекар -> Виж резултат. Минимализмът в интерфейса е нашата конкурентна полза.

2.1.7. Технологични тенденции: AI и Blockchain в медицината

Бъдещето на MedReports включва изкуствен интелект (AI). Интегрирането на LLM (Large Language Models) като GPT-4 през API би позволило на системата автоматично да "превежда" сложните медицински термини в диагнозите на разбираем за пациента език.

Blockchain пък предлага решение на проблема с доверието. Използването на децентрализиран регистър за Audit логовете би направило невъзможно "изтриването" с обратна дата на медицинска грешка. В MedReports за момента използваме SQL базирани логове, но архитектурата ни позволява лесна вградена миграция към Blockchain технология в бъдеще.

2.1.8. Облачни технологии (Cloud Computing) и SaaS модели

MedReports е замислен като SaaS (Software as a Service) продукт. Това позволява на лекарите да ползват системата без да купуват сървъри. Предимствата на облака:
- Мащабируемост: При увеличаване на броя на пациентите, добавяме мощност с един клик.
- Автоматичен бекъп: Данните се дублират в три различни географски точки на Azure/AWS.
- Версионна поддръжка: Всички потребители ползват последната версия на софтуера едновременно.

2.1.9. Обосновка на избора на технологичен стек (.NET 8.0/C#)

Изборът на .NET 8.0 (последната LTS версия) се базира на няколко критични фактора:
- Производителност (Performance): .NET 8 е в Топ 3 на най-бързите уеб рамки в света според класацията TechEmpower.
- Безопасност на типовете (Type Safety): C# е силно типизиран език, което елиминира 80% от логическите грешки в рантайма. В медицината грешка в поле за кръвна група е критична.
- Крос-платформеност: Приложението се компилира и работи еднакво добре на Windows и Linux (Docker).

2.1.10. Предимства на MSIL архитектурата и CLR средата

C# кодът на MedReports не се компилира до машинен код директно. Той става Microsoft Intermediate Language (MSIL). Това позволява на системата да извършва JIT (Just-In-Time) компилация, оптимизирана за конкретния хардуер на сървъра. Common Language Runtime (CLR) пък се грижи за управлението на паметта чрез Garbage Collection, което е жизненоважно за 24/7 системи, които не трябва да се рестартират заради "Memory Leaks".

2.1.11. Криптографски методи за защита на данните (PBKDF2/TLS)

Сигурността в MedReports е на няколко нива:
- Пароли: Използваме Argon2 или PBKDF2 with HMAC-SHA256. Паролата никога не се записва в базата. Записва се нейният хеш с уникална сол за всеки потребител.
- Трафик: Пълна принудителна пренасочваща логика към HTTPS. Изключване на стари и несигурни протоколи като SSL 3.0.
- База данни: Чувствителните полета могат да бъдат шифровани на ниво колона чрез TDE (Transparent Data Encryption).

2.1.12. Облачни технологии и микросървисна готовност (Docker)

Контейнеризацията с Docker е стандарт в съвременния свят. MedReports се разгръща като "Linux Image", което ни гарантира, че библиотеките на сървъра са идентични с тези на разработчика. Това ускорява цикъла на разработка и прави "Deployment"-а скучна и сигурна процедура, а не стресово събитие.

2.2. Глава 2 Изисквания към програмния продукт

2.2.1. Функционални изисквания по потребителски роли (FR)

Функционалните изисквания описват действията, които системата трябва да позволява. Те са сърцето на потребителското изживяване.

Роля "Гост (Guest)":
- FR1: Преглед на публичното съдържание. Потребителят трябва да види мисията на MedReports и предимствата на платформата без да се логва.
- FR2: Секция за контакти. Достъп до форма за запитване с анти-спам защита.
- FR3: Регистрация. Процесът трябва да е лесен чрез имейл и потвърждение на паролата.

Роля "Пациент (Patient)":
- FR4: Управление на профила. Възможност за смяна на аватар, име и координати за връзка.
- FR5: Лично досие. Преглед на цялата хронология на прегледите в "Timeline" изглед.
- FR6: Качване на документи. Пациентът трябва да може да прикачва PNG/JPG/PDF файлове към досието си.
- FR7: Заявка за статус "Лекар". Ако потребителят е медик, той трябва да може да изпрати сертификатите си за одобряване.
- FR8: Езикови настройки. Смяна на езика на интерфейса в движение.

Роля "Лекар (Doctor)":
- FR9: Търсене по идентификатор. Намиране на пациент по имейл адрес или QR код (бъдеща опция).
- FR10: Създаване на медицински запис. Форма за въвеждане на Диагноза (ICD-10 стандартизирана), описание на терапия и препоръки.
- FR11: Управление на "Списък с пациенти". Бърз достъп до последните прегледани лица.

Роля "Администратор (Admin)":
- FR12: Верификация на лекари. Преглед на пратените документи и бутон "Одобри", който сменя ролята в базата.
- FR13: Управление на запитвания. Отговор към съобщенията от контактната форма.
- FR14: Мониторинг на системата. Преглед на броя активни потребители и системни грешки.

2.2.4. Нефункционални изисквания (NFR) и QoS (Quality of Service)

NFR описват как системата се държи, а не какво прави:
- NFR1 (Производителност): Всяко зареждане на страница не трябва да отнема повече от 1.5 секунди при 4G връзка.
- NFR2 (Устойчивост): Системата трябва да поддържа 1000 едновременни сесии без деградация на времето за отговор.
- NFR3 (Наличност): 99.9% Up-time (не повече от 9 часа офлайн на година).
- NFR4 (Консистентност): Всяка транзакция трябва да е ACID compliant.

2.2.5. Потребителски сценарии (User Personas) и психологически профили

2.2.6. Регулаторни изисквания и GDPR съответствие

Порталът трябва да следва "Principle of Least Privilege". Лекарят не вижда "Личния дневник" на пациента, а само клиничните факти. Данните са изолирани чрез "Logic Security" - ID на потребителя се проверява на ниво база срещу всяка заявка, за да се избегне IDOR атака (Insecure Direct Object Reference). Всяко действие се логва с IP адрес и времеви отпечатък.

2.2.7. Психология на интерфейсите и UI/UX стандарти

Използваме "F-Pattern" за четене на информацията. Най-важните бутони (Help, Profile, Record) са в горния десен и горен десен ъгъл. Цветовете следват психологията на доверието: Синьо (спокойствие), Зелено (здраве), Бяло (стерилност и чистота). Избягваме червения цвят, освен при критични индикации (алергии). Интерфейсът е проектиран така, че да не претоварва потребителя с повече от 7-9 елемента на един екран (Millers Law).

2.2.8. Архитектурни изисквания: Мащабируемост, консистентност и надеждност

Проектът MedReports е проектиран да отговаря на високите критерии за мащабируемост (Scalability) и отказоустойчивост (Resilience):

А. Хоризонтално мащабиране: Благодарение на "Stateless" архитектурата (сесиите се съхраняват в шифровани куки или външни услуги), можем да стартираме неограничен брой копия на портала зад Load Balancer (като Azure Traffic Manager или Nginx). 
Б. Оптимизация на заявките: Използването на "Asynchronous Programming" (async/await) в .NET позволява на сървъра да обслужва хиляди едновременни заявки без да блокира системни нишки. Индексираните полета в базата данни (Email, ID) гарантират светкавична скорост на търсене.
В. Надеждност на данните: Базата данни поддържа ACID транзакции. Интегрираният механизъм "Optimistic Concurrency Control" предотвратява загуба на данни при едновременно редактиране на досие от двама лекари чрез скрит маркер за версия (RowVersion).

2.2.9. Обосновка на нормализацията (BCNF срещу 3NF)

За да се гарантира интегритет на здравните данни, базата данни е приведена в Трета нормална форма (3NF). 
- Първа нормална форма (1NF): Всяка клетка от таблицата съдържа атомарни стойности (няма вложени обекти).
- Втора нормална форма (2NF): Всички неключови атрибути зависят функционално от целия основен ключ. Например, детайлите на пациента се съхраняват в неговата таблица, а не в записите на лекаря.
- Трета нормална форма (3NF): Премахнати са транзитивните зависимости. Информацията за медицинския център е в отделна субектна област.

Изборът на 3NF гарантира висока производителност при работа с милиони записи, намалявайки броя на сложните "Joins" в SQL запитванията, което е критично за бързодействието на платформата.

2.3. Глава 3 Проектен цикъл и техническа имплементация

2.3.1. Архитектура на базата данни: Инфо-логически модел

Базата данни на MedReports е проектирана чрез подхода "Code-First", използвайки Entity Framework Core (EF Core). Този метод позволява на разработчика да дефинира структурата на данните чрез C# класове (Models), които след това автоматично се превръщат в релационни таблици в SQL Server чрез процеса на миграция. 

Инфо-логическият модел е изграден върху релационната парадигма, като се обръща специално внимание на връзките "Едно към Много" (1:N) и "Едно към Едно" (1:1). Например, един `User` има точно един `MedicalRecord` (1:1), но един `MedicalRecord` може да съдържа неограничен брой `MedicalEntries` (1:N). Този модел гарантира, че медицинската информация е силно свързана с конкретния субект, но позволява гъвкавост при натрупването на исторически данни.

2.3.2. Каталог на базата данни – описание на таблици и полета

По-долу е изложен пълният каталог на основните субекти в системата:

А. Таблица `Users` (Потребители):
- `Id` (INT, PK): Уникален идентификатор, автоматично нарастващ.
- `Email` (NVARCHAR(256), Unique Index): Служи за вход и е основен идентификатор за търсене на пациент.
- `PasswordHash` (NVARCHAR(MAX)): Съхранява сигурен отпечатък на паролата.
- `Name` (NVARCHAR(100)): Пълното име на потребителя.
- `Role` (NVARCHAR(50)): Текущата роля (Patient, Doctor, Admin).
- `CreatedAt` (DATETIME): Дата и час на създаване на акаунта.

Б. Таблица `MedicalRecords` (Медицински досиета):
- `Id` (INT, PK): Идентификатор на досието.
- `UserId` (INT, FK): Връзка към таблица `Users`.
- `BloodType` (NVARCHAR(5)): Кръвна група (напр. "A+", "B-").
- `Allergies` (NVARCHAR(MAX)): Текстово поле за описване на алергии.
- `ChronicConditions` (NVARCHAR(MAX)): Хронични заболявания на пациента.

В. Таблица `MedicalEntries` (Клинични записи):
- `Id` (INT, PK): Код на записа.
- `MedicalRecordId` (INT, FK): Към кое досие принадлежи този запис.
- `Type` (INT/ENUM): Тип на записа (Диагноза, Терапия, Рецепта, Операция).
- `Title` (NVARCHAR(200)): Кратко заглавие.
- `Description` (NVARCHAR(MAX)): Подробно медицинско описание.
- `Date` (DATETIME): Кога е направен прегледът.
- `CreatedByDoctorId` (INT): Кой лекар е извършил записа.

Г. Таблица `DoctorApplications` (Кандидатури за лекарски статус):
- `Id` (INT, PK): Идентификатор на заявката.
- `UserId` (INT, FK): Кой потребител кандидатства.
- `MedicalLicensePath` (NVARCHAR(MAX)): Път до сканирания документ на сървъра.
- `Status` (INT/ENUM): Pending, Approved, Rejected.

Д. Таблица `MedicalAuditLogs` (Одит на досието):
- `Id` (INT, PK): Код на лога.
- `MedicalRecordId` (INT, FK): Кое досие е променено.
- `OldSnapshot` (JSON): Състояние на данните преди промяната.
- `NewSnapshot` (JSON): Състояние на данните след промяната.
- `Timestamp` (DATETIME): Точен момент на редакцията.

2.3.3. Архитектура на кода - Service Layer и SOLID принципи

Проектът MedReports следва "Layered Architecture" (Слоеста архитектура), която разделя отговорностите на софтуера:
1. Presentation Layer (Controllers/Views): Грижи се за това какво вижда потребителят.
2. Business Logic Layer (Services): Тук се намира "умът" на приложението — изчисления, верификации и бизнес правила.
3. Data Access Layer (Repositories/DbContext): Директната работа с базата данни.

Прилагането на SOLID принципите е критично. Например, принципът за "Dependency Inversion" позволява на контролерите да работят с интерфейси (`IMedicalRecordService`), а не с конкретни класове. Това прави кода тестваем и лесен за промяна — ако утре решим да променим как записваме данни, просто сменяме реализацията на сървиса, без да пипаме контролерите.

2.3.4. MVC модел и Pattern Matching в C# 12

Моделът MVC (Model-View-Controller) е основата на ASP.NET Core. В MedReports използваме "ViewModels", за да прехвърляме данни от базата към интерфейса. Това е стратегия за сигурност — никога не показваме на фронтенда директни обекти от базата данни, а само това, което е необходимо за конкретната страница.

Използваме новите възможности на C# 12, като "Primary Constructors" и "Pattern Matching", за да направим кода по-кратък и четим. Например, проверката на ролите и типовете медицински записи се извършва чрез `switch` изрази с шаблони, което намалява вероятността от логически грешки.

2.3.5. Бизнес логика: Процесът на верификация на лекари

Процесът на верификация е един от най-критичните в системата. Той следва стриктна логика:
1. Потребителят качва своите сертификати (PDF или Изображение).
2. Системата създава запис в `DoctorApplications` със статус "Pending".
3. Администраторът получава достъп до списък с всички чакащи заявки.
4. При одобрение, сървисът извършва транзакционна операция — променя ролята на потребителя в таблица `Users` и отбелязва заявката като "Approved". 
Използването на SQL Транзакции гарантира, че ако възникне грешка по средата (напр. спрял ток), базата данни ще се "върне" в предишното състояние, предотвратявайки създаването на "полу-лекари" в системата.

2.3.6. Имплементация на интерфейса чрез Razor и Partial Views

Интерфейсът на MedReports е изграден с Razor Engine. Използваме "Partial Views" за компоненти, които се повтарят в системата — например навигационната лента, футъра или картите за медицински записи. Това спестява дублиране на код (DRY - Don't Repeat Yourself). Когато променим дизайна на една карта за преглед, тя се обновява автоматично в целия портал.

Razor страниците са тясно интегрирани с Bootstrap 5 и собствен CSS, което позволява рендирането да става на сървъра. Това е по-сигурно за медицински данни, тъй като логиката за достъп не е изложена в браузъра на клиента (както е при чистите JavaScript рамки).

2.3.7. Система за одитиране и "Снимки" на състоянието (Auditing)

Всеки път, когато лекар или пациент промени важна информация в профила (кръвна група, алергии), системата прави "Medical Snapshot". Използваме библиотеката `System.Text.Json` за сериализация на обекта в JSON формат. Тези снимки се съхраняват в таблицата `MedicalAuditLogs`. Това позволява на администраторите и пациентите да видят "Хронологично дърво" на промените. Ако възникне спор кога е добавена определена алергия, системата може да докаже точния момент и старото състояние.

2.3.8. Дизайнерска система – CSS променливи и Grid системи

Дизайнерската система на MedReports е дефинирана чрез CSS Variables (напр. `--med-primary: #2a52be;`). Това позволява на платформата да поддържа "Тъмна тема" (Dark Mode) само чрез смяна на стойностите на променливите. Навигацията и таблото с досиета са изградени чрез CSS Grid, което гарантира, че информацията ще се подреди правилно както на огромен монитор в лекарски кабинет, така и на малък екран на смартфон.

Използваме "Fluid Typography", което означава, че размерът на шрифта се адаптира плавно спрямо ширината на екрана. Това е особено важно за възрастни пациенти, които често ползват голямо увеличение на страниците.

2.3.9. Localization & Globalization механизми

MedReports е двуезична система (Български и Английски). Реализирахме персонализиран `ITranslationService`, който не разчита на статични ресурси, а на динамичен речник, съхраняван в сесията на потребителя. Когато потребителят избере език, системата презарежда страницата и Middleware компонентът инжектира правилните преводи във всяко текстово поле. Този подход позволява на лекари от различни държави да работят с български пациенти, виждайки интерфейса на познат за тях език.

2.3.10. Валидация и "Sanitization" на качвани файлове

Качването на медицински документи е висок риск за сигурността. MedReports прилага три нива на проверка:
1. Разширение: Проверява се дали файлът е PDF или Изображение.
2. MIME Type: Проверява се "сигнатурата" на файла (Magic Bytes), за да се увери, че файлът не е преименуван вирус (напр. .exe файл, преименуван на .jpg).
3. Secure Naming: Системата преименува файла с уникален GUID и го съхранява в папка, която не е достъпна директно през уеб, а само през специален контролер, проверяващ правата на достъп.

2.3.11. Управление на сигурността: AuthN и AuthZ логика

- Authentication (AuthN): Процес на доказване на самоличност чрез имейл и парола. Използваме Cookie Authentication в ASP.NET Core.
- Authorization (AuthZ): Процес на проверка на правата. Приложихме "Policy-based Authorization", което дефинира строги правила: например политиката "AdminOnly" изисква потребителят да има роля 'Admin'. 
Системата използва Middleware за глобална проверка на достъпа, което гарантира, че ако забравим да сложим атрибут `[Authorize]` на някоя страница, тя пак ще бъде защитена по подразбиране.

2.4.2. Пътеводител за пациента - пълно управление на досие

След регистрация, пациентът попада в своето "Здравно табло".
- Моето досие: Тук пациентът попълва своята кръвна група и алергии.
- История: Виждат се всички клинични записи, добавени от лекари.
- Качване на документи: Пациентът може да качи сканиран амбулаторен лист или снимка от частна лаборатория, която не е в системата.
Всеки пациент може да смени своята парола и езикови предпочитания от настройките на профила.

2.4.3. Инструкции за лекари - работа с клинични данни

Лекарят разполага с функционалност за търсене.
1. Въвежда имейла на пациента.
2. Ако пациентът съществува, се отваря неговото досие.
3. Лекарят натиска "Добави запис" и попълва диагнозата.
4. Системата автоматично отбелязва датата и името на лекаря.
Лекарите имат и достъп до преглед на качените от пациента PDF файлове, което помага при консултация на пациенти със стари хартиени досиета.

2.4.4. Административен контрол и одобряване на роли

Администраторът има достъп до специално меню "Админ панел".
- Кандидатури: Списък с всички потребители, искащи статус "Лекар". Администраторът преглежда прикачения лиценз и натиска "Approve" или "Reject".
- Системни логове: Преглед на одитните записи за промени по медицинските досиета.
Администраторът може също така да блокира акаунти при съмнение за злоупотреба с лични данни.

2.4.5. Техническа поддръжка и Disaster Recovery планове

Системата е проектирана за лесна поддръжка:
- Бекъп: Препоръчва се ежедневно архивиране на SQL базата данни.
- Логове: Използваме вградения Logger за записване на критични грешки във файлове на сървъра.
- Disaster Recovery: При критичен срив, системата може да бъде разгърната от Docker образ на нов сървър за по-малко от 5 минути.

2.4.6. Таблица с ръчни тестови сценарии (QA Matrix)

2.4.7. Анализ на рисковете и планове за бъдещо развитие

Основните рискове включват:
- Социално инженерство: Лекари да споделят паролите си. Решение: Въвеждане на двуфакторна аутентикация (2FA) в следваща версия.
- Кражба на данни: Шифроване на диска на сървъра.
Бъдещо развитие: Интеграция с носими устройства (Smart Watches) за автоматично следене на пулс и кръвно налягане, което да се добавя в "Личния дневник" на пациента.


2.4.8. Анализ на сигурността съответствие с OWASP Top 10

Системата MedReports е подложена на одит спрямо десетте най-чести уязвимости в уеб приложенията:
- A01:2021-Broken Access Control: Използваме гъвкава система от "Policies" и мощен Middleware, който проверява всеки опит за достъп до данни в реално време. Пациентите не могат да виждат данни на други пациенти, дори ако ръчно променят ID в адресната лента.
- A02:2021-Cryptographic Failures: Всички пароли са хеширани с PBKDF2 (Password-Based Key Derivation Function 2) със сол. Всички комуникации се извършват през TLS 1.3 за криптиране на данните при пренос.
- A03:2021-Injection:Entity Framework Core автоматично параметризира всички заявки към базата данни, което напълно елиминира риска от SQL инжекции.
- A04:2021-Insecure Design: Прилагането на "Clean Architecture" и SOLID принципите гарантира, че сигурността е заложена в самия дизайн на софтуера (Security by Design).
- A05:2021-Security Misconfiguration: Изключване на подробните съобщения за грешки (Detailed Errors) в продукционна среда за избягване на изтичане на системна информация.
- A06:2021-Vulnerable and Outdated Components: Проектът използва .NET 8.0 - версия с дългосрочна поддръжка (LTS), като всички външни пакети са актуализирани до последните стабилни версии.
- A07:2021-Identification and Authentication Failures: Сложни изисквания за парола (дължина, специални символи) и лимитиране на опитите за вход срещу Brute Force атаки.
- A08:2021-Software and Data Integrity Failures: Всяко качване на файл се проверява хистограма и контролна сума.
- A09:2021-Security Logging and Monitoring: MedicalAuditLogs предоставя пълна визуална картина на всяка промяна в системата.
- A10:2021-Server-Side Request Forgery: Използване на "Allow-list" за разрешени домейни при външни интеграции.

// --- MedicalRecordService.cs (Condensed Logic) ---
public class MedicalRecordService {
    private readonly ApplicationDbContext _db;
    public MedicalRecordService(ApplicationDbContext db) { _db = db; }

    public MedicalRecord? GetRecordByUserId(int userId)
        => _db.MedicalRecords.Include(r => r.Entries.Where(e => !e.IsDeleted))
              .FirstOrDefault(r => r.UserId == userId);

    public MedicalEntry AddEntry(int userId, MedicalEntryType type, string title, string? desc, DateTime date, int drId, string drName) {
        var record = GetRecordByUserId(userId) ?? CreateEmptyRecord(userId);
        var entry = new MedicalEntry { MedicalRecordId = record.Id, Type = type, Title = title, Description = desc, Date = date, CreatedByDoctorId = drId, CreatedByDoctorName = drName };
        _db.MedicalEntries.Add(entry);
        _db.SaveChanges();
        AddAuditLog(record.Id, entry.Id, "AddEntry", drId, drName, $"Added {type}: {title}");
        return entry;
    }

    public void UpdateRecordInfo(int userId, string? blood, string? allergies, string? chronic, int actorId, string actorName) {
        var record = _db.MedicalRecords.FirstOrDefault(r => r.UserId == userId);
        if (record == null) return;
        var oldSnap = JsonSerializer.Serialize(new { record.BloodType, record.Allergies, record.ChronicConditions });
        record.BloodType = blood; record.Allergies = allergies; record.ChronicConditions = chronic;
        _db.SaveChanges();
        var newSnap = JsonSerializer.Serialize(new { record.BloodType, record.Allergies, record.ChronicConditions });
        AddAuditLogWithSnapshot(record.Id, null, "UpdateRecord", actorId, actorName, "Updated info", oldSnap, newSnap);
    }
}

// --- UserService.cs (Identity Logic) ---
public class UserService {
    private readonly ApplicationDbContext _db;
    public UserService(ApplicationDbContext db) { _db = db; }

    public User? GetUserByEmail(string email) => _db.Users.FirstOrDefault(u => u.Email == email);
    public User? Register(string name, string email, string pass) {
        if (GetUserByEmail(email) != null) return null;
        var user = new User { FullName = name, Email = email, Password = HashPassword(pass), CreatedAt = DateTime.Now, Role = UserRole.User };
        _db.Users.Add(user); _db.SaveChanges();
        return user;
    }
    public User? Login(string email, string pass) {
        var user = GetUserByEmail(email);
        return (user != null && VerifyPassword(pass, user.Password)) ? user : null;
    }
    private string HashPassword(string p) => BCrypt.Net.BCrypt.HashPassword(p);
    private bool VerifyPassword(string p, string h) => BCrypt.Net.BCrypt.Verify(p, h);
}

// --- ApplicationDbContext.cs (Data Schema) ---
public class ApplicationDbContext : DbContext {
    public ApplicationDbContext(DbContextOptions options) : base(options) { }
    public DbSet<User> Users { get; set; }
    public DbSet<MedicalRecord> MedicalRecords { get; set; }
    public DbSet<MedicalEntry> MedicalEntries { get; set; }
    public DbSet<MedicalAuditLog> MedicalAuditLogs { get; set; }
    public DbSet<DoctorApplication> DoctorApplications { get; set; }

    protected override void OnModelCreating(ModelBuilder mb) {
        mb.Entity<User>().HasIndex(u => u.Email).IsUnique();
        mb.Entity<MedicalRecord>().HasOne<User>().WithOne().HasForeignKey<MedicalRecord>(r => r.UserId);
    }
}

public class DoctorService {
    private readonly ApplicationDbContext _db;
    public DoctorService(ApplicationDbContext db) { _db = db; }

    public async Task<bool> Apply(int userId, string about, IFormFile license) {
        var path = await SaveFile(license);
        var app = new DoctorApplication { UserId = userId, About = about, MedicalLicensePath = path, Status = ApplicationStatus.Pending };
        _db.DoctorApplications.Add(app);
        return await _db.SaveChangesAsync() > 0;
    }

    public async Task<bool> Approve(int appId, int adminId) {
        var app = await _db.DoctorApplications.FindAsync(appId);
        if (app == null) return false;
        var user = await _db.Users.FindAsync(app.UserId);
        if (user == null) return false;
        user.Role = UserRole.Doctor; app.Status = ApplicationStatus.Approved;
        return await _db.SaveChangesAsync() > 0;
    }
}

public class AuthMiddleware {
    private readonly RequestDelegate _next;
    public AuthMiddleware(RequestDelegate next) { _next = next; }

    public async Task InvokeAsync(HttpContext context) {
        var path = context.Request.Path.Value?.ToLower();
        var role = context.Session.GetString("UserRole");

        if (path != null && path.Contains("/admin") && role != "Admin") {
            context.Response.Redirect("/Home/AccessDenied"); return;
        }
        if (path != null && path.Contains("/medicalrecord/add") && role != "Doctor") {
            context.Response.Redirect("/Home/AccessDenied"); return;
        }
        await _next(context);
    }
}

// --- UserService.cs (Identity Management) ---
public class UserService {
    private readonly ApplicationDbContext _db;
    public UserService(ApplicationDbContext db) { _db = db; }

    public User? GetUserByEmail(string email) => _db.Users.FirstOrDefault(u => u.Email == email);
    public User? Register(string name, string email, string pass) {
        if (GetUserByEmail(email) != null) return null;
        var user = new User { FullName = name, Email = email, Password = HashPassword(pass), Role = UserRole.User };
        _db.Users.Add(user); _db.SaveChanges();
        return user;
    }
    public User? Login(string email, string pass) {
        var user = GetUserByEmail(email);
        return (user != null && VerifyPassword(pass, user.Password)) ? user : null;
    }
    private string HashPassword(string p) => BCrypt.Net.BCrypt.HashPassword(p);
    private bool VerifyPassword(string p, string h) => BCrypt.Net.BCrypt.Verify(p, h);
}

// --- ApplicationDbContext.cs (EF Core Schema) ---
public class ApplicationDbContext : DbContext {
    public ApplicationDbContext(DbContextOptions options) : base(options) { }
    public DbSet<User> Users { get; set; }
    public DbSet<MedicalRecord> MedicalRecords { get; set; }
    public DbSet<MedicalEntry> MedicalEntries { get; set; }
    public DbSet<MedicalAuditLog> MedicalAuditLogs { get; set; }
    public DbSet<DoctorApplication> DoctorApplications { get; set; }

    protected override void OnModelCreating(ModelBuilder mb) {
        mb.Entity<User>().HasIndex(u => u.Email).IsUnique();
        mb.Entity<MedicalRecord>().HasOne<User>().WithOne().HasForeignKey<MedicalRecord>(r => r.UserId);
        mb.Entity<MedicalEntry>().HasIndex(e => e.MedicalRecordId);
    }
}

public class DoctorController : Controller {
    private readonly DoctorService _doctorService;
    public DoctorController(DoctorService dS) { _doctorService = dS; }

    [HttpGet]
    public IActionResult PendingApplications() {
        if (HttpContext.Session.GetString("UserRole") != "Admin") return RedirectToAction("AccessDenied", "Home");
        var apps = _doctorService.GetPendingApps();
        return View(apps);
    }

    [HttpPost]
    public async Task<IActionResult> Approve(int id) {
        var adminId = HttpContext.Session.GetInt32("UserId") ?? 0;
        await _doctorService.Approve(id, adminId);
        return RedirectToAction("PendingApplications");
    }

    [HttpPost]
    public async Task<IActionResult> Reject(int id, string comment) {
        await _doctorService.Reject(id, comment);
        return RedirectToAction("PendingApplications");
    }
}
